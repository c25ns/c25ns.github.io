import pygame # Initialize pygame pygame.init() # Create the screen screen = pygame.display.set_mode((500, 500)) # Create the box box = pygame.Rect(250, 250, 50, 50) # Set the speed of the box speed = 5 # Main game loop while True: # Check for events for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() elif event.type == pygame.KEYDOWN: # If the return key is pressed, move the box if event.key == pygame.K_RETURN: box.x += speed box.y += speed # Fill the screen with black screen.fill((0, 0, 0)) # Draw the box on the screen pygame.draw.rect(screen, (255, 255, 255), box) # Update the display pygame.display.flip()
